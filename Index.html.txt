  <script>
    // Global variables
    let currentTab = 'bookingTab';
    
    // Initialize on page load
    document.addEventListener('DOMContentLoaded', function() {
      console.log('Hospital Room Booking System initialized');
      showLoading(true);
      
      // Initialize date/time
      updateDateTime();
      setInterval(updateDateTime, 1000);
      
      // Set default admission date/time to now
      const now = new Date();
      const datetimeLocal = new Date(now.getTime() - now.getTimezoneOffset() * 60000)
        .toISOString()
        .slice(0, 16);
      document.getElementById('admitDate').value = datetimeLocal;
      
      // Set up event listeners
      setupEventListeners();
      
      // Load initial data
      setTimeout(() => {
        loadDashboard();
        loadAvailableRooms();
        loadAllRooms();
        loadPatients();
        showLoading(false);
      }, 500);
      
      // Handle tab changes
      document.querySelectorAll('#mainTabs a').forEach(tab => {
        tab.addEventListener('shown.bs.tab', function(event) {
          currentTab = event.target.getAttribute('href').substring(1);
          console.log('Tab changed to:', currentTab);
        });
      });
      // Load cleaning rooms when management tab is opened
      const manageTab = document.querySelector('a[href="#manageTab"]');
      if (manageTab) {
        manageTab.addEventListener('shown.bs.tab', function() {
          setTimeout(loadCleaningRooms, 500);
        });
      }
      
      // Load cleaning rooms when cleaning tab is clicked
      const cleaningTab = document.querySelector('a[href="#needsCleaningTab"]');
      if (cleaningTab) {
        cleaningTab.addEventListener('shown.bs.tab', function() {
          loadCleaningRooms();
        });
      }
    });

    function debugRoom(roomNo) {
  showLoading(true);
  google.script.run
    .withFailureHandler(function(error) {
      showMessage('Debug Error', 'Error: ' + error.toString());
      showLoading(false);
    })
    .withSuccessHandler(function(response) {
      let message = `<div class="alert ${response.success ? 'alert-success' : 'alert-danger'}">`;
      message += `<h5>${response.success ? '✓ Found' : '✗ Not Found'}: Room ${roomNo}</h5>`;
      message += `<p>${response.message}</p>`;
      
      if (response.roomDetails) {
        message += `<hr><h6>Room Details:</h6>`;
        message += `<pre class="bg-light p-2">${JSON.stringify(response.roomDetails, null, 2)}</pre>`;
      }
      
      if (response.allRooms) {
        message += `<hr><h6>All Rooms in Sheet:</h6>`;
        message += `<div class="table-responsive"><table class="table table-sm">`;
        message += `<thead><tr><th>Row</th><th>Room No</th><th>Type</th><th>Value</th></tr></thead>`;
        message += `<tbody>`;
        response.allRooms.forEach(room => {
          message += `<tr><td>${room.row}</td><td>${room.roomNo}</td><td>${room.type}</td><td>"${room.value}"</td></tr>`;
        });
        message += `</tbody></table></div>`;
      }
      
      if (response.headers) {
        message += `<hr><h6>Headers Found:</h6>`;
        message += `<p>${response.headers.join(', ')}</p>`;
      }
      
      message += `</div>`;
      showMessage('Debug Results', message);
      showLoading(false);
    })
    .debugRoomSearch(roomNo);
}

    // Show/hide loading spinner
    function showLoading(show) {
      const spinner = document.getElementById('loadingSpinner');
      const mainContent = document.getElementById('mainContent');
      if (show) {
        spinner.style.display = 'block';
        mainContent.style.display = 'none';
      } else {
        spinner.style.display = 'none';
        mainContent.style.display = 'block';
      }
    }

    // Update current date/time
    function updateDateTime() {
      const now = new Date();
      const options = { 
        weekday: 'long', 
        year: 'numeric', 
        month: 'long', 
        day: 'numeric',
        hour: '2-digit',
        minute: '2-digit',
        second: '2-digit'
      };
      document.getElementById('currentDateTime').textContent = 
        now.toLocaleDateString('en-US', options);
    }

    // Setup event listeners
    function setupEventListeners() {
      // Booking form
      document.getElementById('bookingForm').addEventListener('submit', function(e) {
        e.preventDefault();
        bookRoom();
      });
      
      // Checkout form
      document.getElementById('checkoutForm').addEventListener('submit', function(e) {
        e.preventDefault();
        checkOutPatient();
      });
      
      // Filter events
      document.getElementById('filterRoomType').addEventListener('change', loadAvailableRooms);
      document.getElementById('filterCapacity').addEventListener('change', loadAvailableRooms);
    }
    
    // Load dashboard statistics
    function loadDashboard() {
      showLoading(true);
      google.script.run
        .withFailureHandler(function(error) {
          console.error('Dashboard load error:', error);
          displayStats({
            totalRooms: 0,
            occupiedRooms: 0,
            availableRooms: 0,
            needsCleaning: 0,
            totalPatients: 0,
            admittedPatients: 0,
            occupancyRate: 0
          });
          showLoading(false);
        })
        .withSuccessHandler(displayStats)

function displayAllRooms(rooms) {
  const container = document.getElementById('allRoomsTable');
  
  if (!rooms || rooms.length === 0) {
    container.innerHTML = `
      <tr>
        <td colspan="8" class="text-center text-muted py-4">
          <i class="fas fa-bed fa-2x mb-3"></i>
          <p>No rooms found. Add your first room using the "Add New Room" button.</p>
        </td>
      </tr>`;
    showLoading(false);
    return;
  }
  
  let html = '';
  rooms.forEach(room => {
    const statusClass = room.status === 'Occupied' || room.patientCount > 0 ? 
      'badge bg-danger' : 'badge bg-success';
    const cleaningClass = room.cleaningStatus === 'Needs Cleaning' ? 
      'badge bg-warning text-dark' : 'badge bg-success';
    
    // Show patient IDs (comma separated if multiple)
    const patientDisplay = room.currentPatientIds && room.currentPatientIds.length > 0 ? 
      room.currentPatientIds.join(', ') : '<span class="text-muted">-</span>';
    
    // Show occupancy status
    const occupancyStatus = room.patientCount > 0 ? 
      `Occupied (${room.patientCount}/${room.capacity})` : 'Available';
    
    html += `
      <tr>
        <td><strong class="text-primary">${room.roomNo}</strong></td>
        <td>${room.roomType}</td>
        <td>
          <span class="badge bg-dark">${room.capacity} <i class="fas fa-user"></i></span>
          ${room.patientCount > 0 ? 
            `<br><small class="text-muted">(${room.capacity - room.patientCount} left)</small>` : ''}
        </td>
        <td><span class="${statusClass}">${occupancyStatus}</span></td>
        <td>${patientDisplay}</td>
        <td>${room.assignedDate || '<span class="text-muted">-</span>'}</td>
        <td><span class="${cleaningClass}">${room.cleaningStatus}</span></td>
        <td>
          ${room.patientCount === 0 ? `
            <div class="btn-group btn-group-sm">
              <button class="btn btn-outline-success" onclick="markRoomClean('${room.roomNo}')"
                      title="Mark as Clean">
                <i class="fas fa-broom"></i>
              </button>
            </div>
          ` : `
            <button class="btn btn-sm btn-outline-secondary" disabled
                    title="Room has patients">
              <i class="fas fa-ban"></i>
            </button>
          `}
        </td>
      </tr>
    `;
  });
  
  container.innerHTML = html;
  showLoading(false);
}

    // Book room function
    function bookRoom() {
      const fullName = document.getElementById('fullName').value.trim();
      const age = document.getElementById('age').value.trim();
      const gender = document.getElementById('gender').value;
      const roomNo = document.getElementById('roomNo').value;
      
      if (!fullName || !age || !gender || !roomNo) {
        showMessage('Validation Error', 
          'Please fill all required fields:<br>' +
          '• Full Name<br>' +
          '• Age<br>' +
          '• Gender<br>' +
          '• Room Selection<br>' +
          '• Admission Date');
        return;
      }
      
      const patientData = {
        fullName: fullName,
        age: parseInt(age),
        gender: gender,
        condition: document.getElementById('condition').value.trim(),
        doctor: document.getElementById('doctor').value.trim(),
        admitDate: document.getElementById('admitDate').value,
        roomNo: roomNo
      };
      
      if (!confirm(`Confirm booking for ${patientData.fullName} in Room ${patientData.roomNo}?`)) {
        return;
      }
      
      showLoading(true);
      google.script.run
        .withFailureHandler(function(error) {
          console.error('Booking error:', error);
          showMessage('Booking Failed', 'Error: ' + error.toString());
          showLoading(false);
        })
        .withSuccessHandler(handleBookingResponse)
        .bookRoom(patientData);
    }

    // Check out patient
    function checkOutPatient() {
      let patientId = document.getElementById('checkoutPatientId').value.trim();
      if (!patientId) {
        showMessage('Error', 'Please enter Patient ID');
        return;
      }
      
      // Add 'P' prefix if not present
      if (!patientId.startsWith('P')) {
        patientId = 'P' + patientId;
      }
      
      performCheckout(patientId);
    }

    function quickCheckout(patientId) {
      performCheckout(patientId);
    }

    function performCheckout(patientId) {
      if (!confirm(`Are you sure you want to discharge patient ${patientId}?\n\nThis will: \n• Mark patient as discharged \n• Free up the room \n• Mark room for cleaning`)) {
        return;
      }
      
      showLoading(true);
      google.script.run
        .withFailureHandler(function(error) {
          console.error('Checkout error:', error);
          showMessage('Discharge Failed', 'Error: ' + error.toString());
          showLoading(false);
        })
        .withSuccessHandler(handleCheckoutResponse)
        .checkOutPatient(patientId);
    }

function handleBookingResponse(response) {
      if (response.success) {
        showMessage('Success', `Room booked successfully!<br>Patient ID: ${response.patientId}<br>Booking ID: ${response.bookingId}`);
        document.getElementById('bookingForm').reset();
        document.getElementById('admitDate').value = new Date().toISOString().split('T')[0];
        loadDashboard();
        loadRooms();
        loadPatients();
      } else {
        showMessage('Error', response.message);
      }
    }

    // Check out patient function
    function checkOutPatient(e) {
      e.preventDefault();
      
      const patientId = document.getElementById('checkoutPatientId').value;
      if (!patientId) {
        showMessage('Error', 'Please enter Patient ID');
        return;
      }
      
      if (confirm(`Are you sure you want to check out patient ${patientId}?`)) {
        google.script.run.withSuccessHandler(handleCheckoutResponse).checkOutPatient(patientId);
      }
    }

    function handleCheckoutResponse(response) {
      if (response.success) {
        showMessage('Success', response.message);
        document.getElementById('checkoutForm').reset();
        loadDashboard();
        loadRooms();
        loadPatients();
      } else {
        showMessage('Error', response.message);
      }
    }

// Update handleCheckoutResponse to show capacity info
function handleCheckoutResponse(response) {
  if (response.success) {
    let statusMessage = '';
    if (response.remainingPatients > 0) {
      statusMessage = `<p><strong>Room ${response.roomNo}:</strong> Still has ${response.remainingPatients} patient(s)</p>`;
    } else {
      statusMessage = `<p><strong>Room ${response.roomNo}:</strong> Now empty and marked for cleaning</p>`;
    }
    
    showMessage('Patient Discharged', 
      `<div class="alert alert-success">
        <h5><i class="fas fa-check-circle me-2"></i> Discharge Successful!</h5>
        <hr>
        <p>${response.message}</p>
        ${statusMessage}
        <p><strong>Discharge Time:</strong> ${response.timestamp}</p>
      </div>`);
    
    document.getElementById('checkoutPatientId').value = '';
    
    // Refresh all data
    setTimeout(() => {
      loadDashboard();
      loadAvailableRooms();
      loadAllRooms();
      loadPatients();
      if (currentTab === 'manageTab') {
        loadCleaningRooms();
      }
    }, 1000);
    
  } else {
    showMessage('Discharge Failed', 
      `<div class="alert alert-danger">
        <h5><i class="fas fa-exclamation-triangle me-2"></i> Discharge Error</h5>
        <p>${response.message}</p>
      </div>`);
  }
  showLoading(false);
}
